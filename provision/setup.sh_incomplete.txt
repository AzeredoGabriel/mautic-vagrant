
echo "Instalando apache2 e git \n"
sudo apt-get update
sudo apt-get install apache2 git -y

echo "Liberando permissão 755 para pasta /var/www/ \n"
sudo chmod -R 755 /var/www

echo "Configurando GIT com o usuário $git_user e e-mail $git_mail"
git config --global user.name $git_user
git config --global user.email $git_mail

echo "Instalando PHP7.0 e extensões \n"
sudo add-apt-repository ppa:ondrej/php -y
echo -en "\n\n"
sudo apt-get install -y language-pack-en-base
sudo LC_ALL=en_US.UTF-8 add-apt-repository ppa:ondrej/php
sudo apt-get update
sudo apt-get install php7.0 php7.0-mysql -y
sudo apt-get install libapache2-mod-php7.0 php7.0-mcrypt php7.0-bcmath php7.0-mbstring php7.0-curl php7.0-xml php7.0-zip -y

echo "Instalando composer \n"
sudo curl -sS https://getcomposer.org/installer | sudo php -- --install-dir=/usr/local/bin --filename=composer -y

echo "Instalando mysql server e criando senha padrão para o usuário root definida no arquivo de senha \n"
sudo debconf-set-selections <<< 'mysql-server mysql-server/root_password password $mysql_root_password'
sudo debconf-set-selections <<< 'mysql-server mysql-server/root_password_again password $mysql_root_password'
sudo apt-get install mysql-server -y

sudo mysql_install_db
sudo mysql_secure_installation

echo "Criando usuário $mysql_mautic_user no mysql, com a senha $mysql_mautic_password, com acesso ao banco de dados $mysql_mautic_database \n"
mysql --user=root --password=admin --execure="CREATE DATABASE $mysql_mautic_database; GRANT ALL PRIVILEGES ON $mysql_mautic_database.* TO '$mysql_mautic_user'@'%' IDENTIFIED BY '$mysql_mautic_password';"

cd /var/www/

echo "Baixando mautic \n"
git clone $url_mautic_github
#git clone https://github.com/mautic/mautic.git

echo "Ativando mautic no apache"
sudo a2dissite 000-default.conf
sudo a2ensite mautic.conf
